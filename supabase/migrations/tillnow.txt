create table public.caregiver_reviews (
  id uuid not null default gen_random_uuid (),
  booking_id uuid not null,
  caregiver_id uuid not null,
  reviewer_id uuid not null,
  rating numeric(2, 1) not null,
  comment text null,
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  updated_at timestamp with time zone not null default timezone ('utc'::text, now()),
  constraint caregiver_reviews_pkey primary key (id),
  constraint caregiver_reviews_booking_id_key unique (booking_id),
  constraint caregiver_reviews_booking_id_fkey foreign KEY (booking_id) references plant_sitting_bookings (id) on delete CASCADE,
  constraint caregiver_reviews_caregiver_id_fkey foreign KEY (caregiver_id) references caregivers (id) on delete CASCADE,
  constraint caregiver_reviews_reviewer_id_fkey foreign KEY (reviewer_id) references profiles (id) on delete CASCADE,
  constraint caregiver_reviews_rating_check check (
    (
      (rating >= (1)::numeric)
      and (rating <= (5)::numeric)
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_caregiver_reviews_caregiver_id on public.caregiver_reviews using btree (caregiver_id) TABLESPACE pg_default;

create index IF not exists idx_caregiver_reviews_reviewer_id on public.caregiver_reviews using btree (reviewer_id) TABLESPACE pg_default;

create trigger refresh_caregiver_review_stats_trigger
after INSERT
or DELETE
or
update on caregiver_reviews for EACH row
execute FUNCTION handle_caregiver_review_change ();

create trigger set_caregiver_reviews_updated_at BEFORE
update on caregiver_reviews for EACH row
execute FUNCTION set_updated_at ();






create table public.caregivers (
  id uuid not null,
  bio text null,
  price_per_day numeric(10, 2) not null default 0,
  years_experience integer not null default 0,
  location text null,
  specialties text[] not null default '{}'::text[],
  is_available boolean not null default true,
  rating numeric(2, 1) not null default 0,
  review_count integer not null default 0,
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  updated_at timestamp with time zone not null default timezone ('utc'::text, now()),
  constraint caregivers_pkey primary key (id),
  constraint caregivers_id_fkey foreign KEY (id) references profiles (id) on delete CASCADE,
  constraint caregivers_price_per_day_check check ((price_per_day >= (0)::numeric)),
  constraint caregivers_rating_check check (
    (
      (rating >= (0)::numeric)
      and (rating <= (5)::numeric)
    )
  ),
  constraint caregivers_review_count_check check ((review_count >= 0)),
  constraint caregivers_years_experience_check check ((years_experience >= 0))
) TABLESPACE pg_default;

create index IF not exists idx_caregivers_available on public.caregivers using btree (is_available) TABLESPACE pg_default;

create trigger set_caregivers_updated_at BEFORE
update on caregivers for EACH row
execute FUNCTION set_updated_at ();

  species text not null,
  price numeric(10, 2) not null,
  category text not null,
  condition text not null,
  description text null,
  city text null,
  image_url text null,
  status text not null default 'active'::text,
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  updated_at timestamp with time zone not null default timezone ('utc'::text, now()),
  phone_number text null,
  image_urls text[] not null default '{}'::text[],

  constraint plant_listings_seller_id_fkey foreign KEY (seller_id) references profiles (id) on delete CASCADE,

      category = any (

          'Tropical'::text,
          'Herbs'::text

      )
    )
  ),

    (
      condition = any (
        array['Excellent'::text, 'Good'::text, 'Fair'::text]
      )
    )
  ),
  constraint plant_listings_price_check check ((price >= (0)::numeric)),
  constraint plant_listings_status_check check (
    (
      status = any (
        array['active'::text, 'sold'::text, 'archived'::text]
      )
    )
  )
) TABLESPACE pg_default;

create table public.plant_sitting_bookings (
  id uuid not null default gen_random_uuid (),
  owner_id uuid not null,
  caregiver_id uuid not null,
  plant_name text null,
  notes text null,
  start_date date not null,
  end_date date not null,
  total_price numeric(10, 2) not null default 0,
  status text not null default 'pending'::text,
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  updated_at timestamp with time zone not null default timezone ('utc'::text, now()),
  constraint plant_sitting_bookings_pkey primary key (id),
  constraint plant_sitting_bookings_caregiver_id_fkey foreign KEY (caregiver_id) references caregivers (id) on delete RESTRICT,
  constraint plant_sitting_bookings_owner_id_fkey foreign KEY (owner_id) references profiles (id) on delete CASCADE,
  constraint booking_date_order check ((end_date >= start_date)),
  constraint plant_sitting_bookings_status_check check (
    (
      status = any (
        array[
          'pending'::text,
          'confirmed'::text,
          'completed'::text,
          'cancelled'::text
        ]
      )
    )
  ),
  constraint plant_sitting_bookings_total_price_check check ((total_price >= (0)::numeric))
) TABLESPACE pg_default;

create index IF not exists idx_bookings_owner_id on public.plant_sitting_bookings using btree (owner_id) TABLESPACE pg_default;

create index IF not exists idx_bookings_caregiver_id on public.plant_sitting_bookings using btree (caregiver_id) TABLESPACE pg_default;

create index IF not exists idx_bookings_status on public.plant_sitting_bookings using btree (status) TABLESPACE pg_default;
